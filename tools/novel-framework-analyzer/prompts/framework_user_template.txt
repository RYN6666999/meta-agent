## 輸出 JSON Schema

請嚴格按照以下 schema 輸出，所有欄位必填（insufficient_context 為 true 時除外）：

```
{
  "scene_id": "{scene_id}",
  "scene_summary": "<100字內的場景摘要，概括核心衝突與走向>",
  "main_characters": [
    {
      "name": "<角色名>",
      "role": "<在此場景的角色定位，如「主動方」「被迫應對者」「見證者」>",
      "stance": "<此場景中的立場或態度>"
    }
  ],
  "situation": {
    "external_situation": "<外部局勢描述>",
    "power_dynamics": "<誰強誰弱，依據為何>",
    "risks_and_constraints": "<風險、限制、不可跨越的邊界>",
    "active_party": "<主動方角色名>",
    "passive_party": "<被動方角色名>",
    "resource_holders": "<掌握關鍵資源者及其資源類型>",
    "insufficient_context": false,
    "evidence": [
      {
        "text": "<原文引用句>",
        "relevance": "<此引用如何支持局勢分析>"
      }
    ]
  },
  "desire": {
    "explicit_desire": "<顯性欲望：角色公開表達或行動展現的訴求>",
    "implicit_desire": "<隱性欲望：未說出口或本人未意識到的真實渴望>",
    "true_objective": "<若顯性與隱性有差距，推斷最深層的驅動力>",
    "desire_conflicts": "<自我衝突或與他人欲望的碰撞>",
    "obstacles": "<阻止欲望實現的障礙>",
    "insufficient_context": false,
    "evidence": [
      {
        "text": "<原文引用句>",
        "relevance": "<此引用如何揭示欲望>"
      }
    ]
  },
  "mind_shift": {
    "before_mindset": "<場景開始時的心態：信念、情緒或策略>",
    "trigger_event": "<觸發轉變的具體事件或對話>",
    "after_mindset": "<場景結束時的新心態>",
    "shift_type": "<emotion|values|strategy|identity|stance|none>",
    "shift_description": "<從 X 到 Y 的轉變描述，包含轉變幅度評估>",
    "is_reversible": true,
    "insufficient_context": false,
    "evidence": [
      {
        "text": "<原文引用句，最好是觸發心變的關鍵對話或動作>",
        "relevance": "<此引用如何展現心態轉變>"
      }
    ]
  },
  "framework_match": {
    "match_level": "<full|partial|weak|none>",
    "reasoning": "<判定理由：哪些維度強，哪些薄弱，為什麼給這個等級>",
    "missing_dimensions": ["<缺失或薄弱的維度名稱，如空陣列表示齊備>"],
    "insufficient_context": false
  },
  "evidence": {
    "strongest_quotes": [
      {
        "text": "<最能支持整體框架判定的原文引用>",
        "supports": "<支持哪個維度的哪個判斷>"
      }
    ]
  },
  "confidence": 0.85,
  "analysis_notes": "<分析者注記：特殊情況、不確定點、建議人工複查的原因>"
}
```
